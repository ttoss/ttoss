import { Accordion } from "@ttoss/components";

This components uses [react-accessible-accordion package](https://github.com/springload/react-accessible-accordion) as a base and apply [@ttoss/ui](/docs/modules/packages/ui/) styles.

## Example

You can check more examples on [accordion stories](https://storybook.ttoss.dev/?path=/story/components-accordion).

```jsx
import { Accordion } from "@ttoss/components";
import { Text } from "@ttoss/ui";

const MyAccordion = () => {
  return (
    <Accordion>
      <Accordion.Item>
        <Accordion.ItemHeading>
          <Accordion.ItemButton>
            What harsh truths do you prefer to ignore?
          </Accordion.ItemButton>
        </Accordion.ItemHeading>
        <Accordion.ItemPanel>
          <Text>
            Exercitation in fugiat est ut ad ea cupidatat ut in cupidatat
            occaecat ut occaecat consequat est minim minim esse tempor laborum
            consequat esse adipisicing eu reprehenderit enim.
          </Text>
        </Accordion.ItemPanel>
      </Accordion.Item>
      <Accordion.Item>
        <Accordion.ItemHeading>
          <Accordion.ItemButton>
            Is free will real or just an illusion?
          </Accordion.ItemButton>
        </Accordion.ItemHeading>
        <Accordion.ItemPanel>
          <Text>
            In ad velit in ex nostrud dolore cupidatat consectetur ea in ut
            nostrud velit in irure cillum tempor laboris sed adipisicing eu esse
            duis nulla non.
          </Text>
        </Accordion.ItemPanel>
      </Accordion.Item>
    </Accordion>
  );
};
```

## Styles

You can style `Accordion` component passing the `sx` prop and each sub-component with the following classes:

- `Accordion.Item`: `accordion__item`
- `Accordion.ItemHeading`: `accordion__heading`
- `Accordion.ItemButton`: `accordion__button`
- `Accordion.ItemPanel`: `accordion__panel`

Example:

```jsx
// ...
<Accordion
  sx={{
    backgroundColor: 'aqua',
    '.accordion__item': {
      backgroundColor: 'yellow',
    },
    '.accordion__heading': {
      backgroundColor: 'red',
    },
    '.accordion__button': {
      backgroundColor: 'blue',
    },
    '.accordion__panel': {
      backgroundColor: 'white',
    },
  }}
>
```

You can also use the variant `theme.accordion` to style on theme.

```ts
const theme = {
  // rest of the theme
  accordion: {
    // accordion styles
    ".accordion__item": {},
    ".accordion__heading": {},
    ".accordion__button": {},
    ".accordion__panel": {},
  },
};
```
